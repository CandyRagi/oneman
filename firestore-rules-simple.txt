# SIMPLE FIRESTORE RULES FOR TESTING
# Copy and paste these rules into Firebase Console > Firestore Database > Rules
# These rules allow all authenticated users to read/write everything (for testing only)

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}

# IMPORTANT: These rules are for development/testing only!
# They allow any authenticated user to read/write any document.
# Use the more restrictive rules in firestore-rules.txt for production.
